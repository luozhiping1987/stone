class Fib {
    fib0 = 0
    fib1 = 1
    def fib (n) {
        如果 n == 0 {
            fib0
        } 不然 {
            如果 n == 1 {
                fib1
            } 不然 {
                fib(n-1) + fib(n-2)
            }
        }
    }
}
t = currentTime()
f = Fib.new
f.fib 20
