类别 Fib {
    fib0 = 0
    fib1 = 1
    定义 fib (n) {
        如果 n == 0 {
            fib0
        } 不然 {
            如果 n == 1 {
                this.fib1
            } 不然 {
                this.fib(n-1) + this.fib(n-2)
            }
        }
    }
}
t = currentTime()
f = Fib.new
f.fib 20
cost = currentTime() - t
